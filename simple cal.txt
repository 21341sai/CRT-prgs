#include <stdio.h>
#include <math.h>

// // Function declarations
// int sum(int x, int y);
// int sub(int x, int y);
// int mul(int x, int y);
// int divide(int x, int y);
// int power(int x, int y);
int sum(int x, int y) {
    return x + y;
}

int sub(int x, int y) {
    return x - y;
}

int mul(int x, int y) {
    return x * y;
}

int divide(int x, int y) {
    return x / y;
}

int power(double x) {
    return pow(x, 2);
}
int sqroot(double x){
    return sqrt(x);
}

int main() {
    int m, n, res, c;
    while(1){
    printf("Choose operation:\n1. Addition\n2. Subtraction\n3. Multiplication\n4. Division\n5. Power\n6.squareroot\n");
    scanf("%d", &c);
  
    switch(c) {
        case 1:
            printf("Enter two integers: ");
            scanf("%d %d", &m, &n);
            res = sum(m, n);
            break;
        case 2:
            printf("Enter two integers: ");
            scanf("%d %d", &m, &n);
            res = sub(m, n);
            break;
        case 3:
            printf("Enter two integers: ");
            scanf("%d %d", &m, &n);
            res = mul(m, n);
            break;
        case 4:
            printf("Enter two integers: ");
            scanf("%d %d", &m, &n);
            if (n != 0) {
                res = divide(m, n);
            } else {
                printf("Error: Division by zero.\n");
                return 1;
            }
            break;
        case 5:
             printf("Enter the number for power ");
            scanf("%d",&m);
            res = power(m);
            break;
        case 6:
             printf("Enter the number for square root");
            scanf("%d",&m);
            res = sqroot(m);
            break;
        default:
            printf("Invalid operation!!\n");
            return 1;
    }

    printf("Result: %d\n", res);
    }
    return 0;
}
